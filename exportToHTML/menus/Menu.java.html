14   public abstract class Menu

16       final Pattern pattern = Pattern.compile("-?\\d+(\\.\\d+)?");
17       
18       public boolean isNumeric(String str)
19       {
20           if(str.equals(null))
21               return false;
22                  
23           /* 
24           -?          determines if number starts with minus 
25           \d+         matches one or more digits 
26           (\.\d+)?    matches for decimal point and digits following 
27           */
28           
29           return pattern.matcher(str).matches();
30       }
31       
32       public static int validateInput(String message)
33       {
34           Scanner scanner = new Scanner(System.in);
35           String inputChoice;
36           int choice = 0;
37           
38           System.out.println(message);
39           inputChoice = scanner.nextLine();
40           try
41           {
42               //wrapping parseInt in try catch in case an int was not entered
43               choice = Integer.parseInt(inputChoice);
44           } 
45           catch (NumberFormatException x)
46           {
47               //provides details on the exception that is thrown and also provides a more concise, readable error message
48               System.out.println("\n" + x + "\n\nYou have entered an illegal value, please provide a valid input\n");
49               return validateInput(message);
50           }
51           return choice;
52       }
