7    import dataStructures.dll.DLL;
8    import java.util.Scanner;
9    import java.time.LocalDate;
10   import java.time.LocalTime;
11   
12   /**
13    *
14    * @author rayan
15    */
16   public class MClubLog extends clubmatcher.ClubMatcher
17   {
18       private int choice = 0;
19       private Scanner scanner = new Scanner(System.in);
20       private String input = "";
21       private String club_name = null;
22       
23       public MClubLog()
24       {
25           System.out.println("\n=============\nCLUB LOG MENU\n=============\n");
26   
27           do
28           {
29               choice = validateInput
30               (
31                   "\n\t 0 - Exit to Main Menu"
32                   + "\n\t 1 - Output the Club Log"
33                   + "\n\t 2 - Sign student(s) into a club"
34                   + "\n\t 3 - Enter Attendance Menu"
35                   + "\n\t 4 - View the top 3 students with the highest club attendance"
36                   + "\n\t 5 - View the top 3 clubs with the highest attendance"
37               );
38               
39               switch(choice)
40               {
41                   case 1:
42                       db.queryOutput("SELECT * FROM ClubLog", "ClubLog");
43                       break;
44                       
45                   case 2:
46                       do
47                       {
48                           System.out.println("Please input the name of the club (enter \"list\" to list the club names stored in the database):");
49                           club_name = scanner.nextLine();
50   
51                           if(club_name.equals("list"))
52                               db.queryOutput("SELECT Club.ClubName FROM Club", "");
53                       }
54                       while(club_name == null || club_name.equals("list"));
55                       
56                       
57                       DLL<String> students = new DLL();
58                       System.out.println
59                       (
60                           "(type \"list\" to list the students in the database)"
61                           + "\ntype \"exit\" or \"e\" when you have inputted all the students you want to add: "
62                       );
63                       input = scanner.nextLine();
64                       while(!(input.equals("exit") || input.equals("e")))
65                       {
66                           if(input.equals("list"))
67                               db.queryOutput("SELECT Person.PersonName FROM Student INNER JOIN Person ON Student.PersonID = Person.PersonID", "");
68                           else
69                               students.append(input);
70                           
71                           System.out.println("Please input a student, type \"list\" to list students or type \"exit\" or \"e\" when you have inputted all the students you want to add: ");
72                           input = scanner.nextLine();
73                       }
74                       
75                       
76                       for(String s : students)
77                       {
78                           db.executeQuery("INSERT INTO ClubLog VALUES('" + LocalTime.now().toString() + "', '" + LocalDate.now().toString() + "', " + db.getID(s, "Student") + ", " + db.getID(club_name, "Club") + ")");
79                       }
80                       break;
81                       
82                   case 3:
83                       MAttendance attendance = new MAttendance();
84                       break;
85                       
86                   case 4:
87                       db.queryOutput
88                       (
89                               "SELECT Person.PersonName "
90                               + "FROM ClubLog "
91                               + "INNER JOIN Student ON Student.StudentID = ClubLog.StudentID "
92                               + "INNER JOIN Person ON Student.PersonID = Person.PersonID "
93                               + "GROUP BY ClubLog.StudentID ORDER BY COUNT(*) DESC LIMIT 3 "
94                               , ""
95                       );
96                       break;
97                       
98                   case 5:
99                       db.queryOutput
100                      (
101                              "SELECT Club.ClubName "
102                              + "FROM ClubLog "
103                              + "INNER JOIN Club ON ClubLog.ClubID = Club.ClubID "
104                              + "GROUP BY ClubLog.ClubID ORDER BY COUNT(*) DESC LIMIT 3"
105                              , ""
106                      );
107              }
108          }
109          while(choice != 0);
110      }
111  }
